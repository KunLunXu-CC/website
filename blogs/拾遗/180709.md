#### 文件上传组件 
```js
// 文件上传
onUpload = (query,file, cb) => {
  let formData = new FormData();
  formData.append("file", file);

  const fetchParams = {
    method: 'POST',
    url: `${this.url}/api/v1/drive/files`,
    query,
    // 不加 headers body参数上传不了
    headers: {
      'Content-Length': file.size,
    },
    body: formData,
    callback(res){
      cb && cb(res);
    }
  }
  fetchHelper(fetchParams, this.commonStort);
}
```
